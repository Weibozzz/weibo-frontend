## 定义和赋值
```js
var name = 'weibo'; // string
let age = 30; // number
const sex = '男'
const PI = '3.1415'
let hobby = ['篮球', '乒乓球'] // array
let mobile = { // object
  color: 'red',
  tel: 156,
  brand: 'apple'
}
let uni = undefined // undefined
let empty = null // null
let isMan = false // boolean
let sm = Symbol() // symbol
```
